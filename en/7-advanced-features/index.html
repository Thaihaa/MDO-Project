Dynamic Dependency Resolver

File: src/advanced_dependency_manager.py

#!/usr/bin/env python3
import yaml
import json
import boto3
from typing import Dict, List, Set
from datetime import datetime, timedelta

class AdvancedDependencyManager:
    def __init__(self, config_path='config/service_dependencies.yaml'):
        with open(config_path, 'r') as f:
            self.config = yaml.safe_load(f)
        
        self.ecs_client = boto3.client('ecs')
        self.cloudwatch = boto3.client('cloudwatch')
    
    def resolve_dynamic_dependencies(self, services: List[str], environment: str) -> Dict:
        """Dynamically resolve dependencies based on current system state"""
        
        # Get current service health status
        health_status = self.get_services_health(services)
        
        # Analyze deployment history for optimization
        deployment_history = self.analyze_deployment_history(services)
        
        # Create optimized deployment plan
        deployment_plan = self.create_optimized_plan(
            services, health_status, deployment_history, environment
        )
        
        return deployment_plan
    
    def get_services_health(self, services: List[str]) -> Dict:
        """Get current health status of all services"""
        health_status = {}
        
        for service in services:
            try:
                # Get ECS service status
                response = self.ecs_client.describe_services(
                    cluster='restaurant-cluster',
                    services=[f'restaurant-{service}']
                )
                
                if response['services']:
                    ecs_service = response['services'][0]
                    running_count = ecs_service['runningCount']
                    desired_count = ecs_service['desiredCount']
                    
                    health_status[service] = {
                        'ecs_healthy': running_count == desired_count,
                        'running_count': running_count,
                        'desired_count': desired_count,
                        'deployment_status': ecs_service['deployments'][0]['status']
                    }
                else:
                    health_status[service] = {'ecs_healthy': False}
                    
            except Exception as e:
                health_status[service] = {'ecs_healthy': False, 'error': str(e)}
        
        return health_status
    
    def analyze_deployment_history(self, services: List[str]) -> Dict:
        """Analyze historical deployment patterns for optimization"""
        history = {}
        
        for service in services:
            # Get CloudWatch metrics for deployment times
            end_time = datetime.utcnow()
            start_time = end_time - timedelta(days=30)
            
            try:
                response = self.cloudwatch.get_metric_statistics(
                    Namespace='Restaurant/Deployment',
                    MetricName='DeploymentDuration',
                    Dimensions=[
                        {'Name': 'ServiceName', 'Value': service}
                    ],
                    StartTime=start_time,
                    EndTime=end_time,
                    Period=3600,
                    Statistics=['Average', 'Maximum', 'Minimum']
                )
                
                if response['Datapoints']:
                    avg_duration = sum(dp['Average'] for dp in response['Datapoints']) / len(response['Datapoints'])
                    max_duration = max(dp['Maximum'] for dp in response['Datapoints'])
                    
                    history[service] = {
                        'average_deployment_time': avg_duration,
                        'max_deployment_time': max_duration,
                        'deployment_count': len(response['Datapoints'])
                    }
                else:
                    # Default values for new services
                    history[service] = {
                        'average_deployment_time': 120,  # 2 minutes default
                        'max_deployment_time': 300,     # 5 minutes default
                        'deployment_count': 0
                    }
                    
            except Exception as e:
                history[service] = {
                    'average_deployment_time': 120,
                    'max_deployment_time': 300,
                    'deployment_count': 0,
                    'error': str(e)
                }
        
        return history
    
    def create_optimized_plan(self, services: List[str], health_status: Dict, 
                            deployment_history: Dict, environment: str) -> Dict:
        """Create an optimized deployment plan based on current conditions"""
        
        # Determine deployment strategy based on environment and health
        if environment == 'production':
            strategy = 'canary_deployment' if any(
                not status.get('ecs_healthy', True) for status in health_status.values()
            ) else 'blue_green'
        else:
            strategy = 'parallel_optimized'
        
        # Create dependency groups with dynamic optimization
        dependency_groups = self.create_dynamic_groups(
            services, health_status, deployment_history
        )
        
        # Estimate deployment times
        estimated_times = self.estimate_deployment_times(
            dependency_groups, deployment_history, strategy
        )
        
        return {
            'strategy': strategy,
            'dependency_groups': dependency_groups,
            'estimated_times': estimated_times,
            'health_status': health_status,
            'deployment_history': deployment_history,
            'optimization_applied': True,
            'created_at': datetime.utcnow().isoformat()
        }
    
    def create_dynamic_groups(self, services: List[str], health_status: Dict, 
                            deployment_history: Dict) -> List[List[str]]:
        """Create deployment groups with dynamic optimization"""
        
        # Base dependencies
        dependencies = {
            'auth-service': [],
            'notification-service': [],
            'order-service': ['auth-service'],
            'payment-service': ['auth-service']
        }
        
        # Sort services by deployment time (faster services first within groups)
        def get_deployment_time(service):
            return deployment_history.get(service, {}).get('average_deployment_time', 120)
        
        groups = []
        remaining_services = set(services)
        
        while remaining_services:
            current_group = []
            
            # Find services with no unmet dependencies
            for service in list(remaining_services):
                service_deps = dependencies.get(service, [])
                if all(dep not in remaining_services for dep in service_deps):
                    current_group.append(service)
            
            if current_group:
                # Sort by deployment time within group
                current_group.sort(key=get_deployment_time)
                groups.append(current_group)
                
                for service in current_group:
                    remaining_services.remove(service)
            else:
                # Fallback for circular dependencies
                break
        
        return groups
    
    def estimate_deployment_times(self, dependency_groups: List[List[str]], 
                                deployment_history: Dict, strategy: str) -> Dict:
        """Estimate deployment times based on strategy and history"""
        
        total_sequential = 0
        total_parallel = 0
        
        for group in dependency_groups:
            group_times = [
                deployment_history.get(service, {}).get('average_deployment_time', 120)
                for service in group
            ]
            
            # Sequential: sum of all times
            total_sequential += sum(group_times)
            
            # Parallel: max time in each group
            total_parallel += max(group_times) if group_times else 0
        
        # Strategy-specific adjustments
        strategy_multipliers = {
            'parallel_optimized': 1.0,
            'canary_deployment': 1.5,  # Longer due to canary validation
            'blue_green': 1.2          # Longer due to traffic switching
        }
        
        multiplier = strategy_multipliers.get(strategy, 1.0)
        
        return {
            'sequential': total_sequential,
            'parallel_optimized': total_parallel * multiplier,
            'strategy_adjusted': total_parallel * multiplier,
            'time_saved': total_sequential - (total_parallel * multiplier),
            'improvement_percentage': ((total_sequential - (total_parallel * multiplier)) / total_sequential) * 100 if total_sequential > 0 else 0
        }

# Usage example
if __name__ == "__main__":
    manager = AdvancedDependencyManager()
    
    services = ['auth-service', 'notification-service', 'order-service', 'payment-service']
    plan = manager.resolve_dynamic_dependencies(services, 'production')
    
    print(json.dumps(plan, indent=2, default=str))

Automated Alerting System

File: scripts/setup_monitoring.py

#!/usr/bin/env python3
import boto3
import json

def setup_comprehensive_monitoring():
    """Setup comprehensive monitoring and alerting"""
    
    cloudwatch = boto3.client('cloudwatch')
    sns = boto3.client('sns')
    
    # Create SNS topic for alerts
    topic_response = sns.create_topic(Name='restaurant-deployment-alerts')
    topic_arn = topic_response['TopicArn']
    
    # Subscribe email to SNS topic (you would add real email)
    sns.subscribe(
        TopicArn=topic_arn,
        Protocol='email',
        Endpoint='devops-team@restaurant.com'
    )
    
    # Create CloudWatch alarms
    alarms = [
        {
            'AlarmName': 'HighDeploymentFailureRate',
            'ComparisonOperator': 'GreaterThanThreshold',
            'EvaluationPeriods': 2,
            'MetricName': 'DeploymentErrors',
            'Namespace': 'Restaurant/Deployment',
            'Period': 300,
            'Statistic': 'Sum',
            'Threshold': 3.0,
            'ActionsEnabled': True,
            'AlarmActions': [topic_arn],
            'AlarmDescription': 'Alert when deployment error rate is high',
            'Unit': 'Count'
        },
        {
            'AlarmName': 'LongDeploymentDuration',
            'ComparisonOperator': 'GreaterThanThreshold',
            'EvaluationPeriods': 1,
            'MetricName': 'DeploymentDuration',
            'Namespace': 'Restaurant/Deployment',
            'Period': 300,
            'Statistic': 'Average',
            'Threshold': 600.0,  # 10 minutes
            'ActionsEnabled': True,
            'AlarmActions': [topic_arn],
            'AlarmDescription': 'Alert when deployment takes too long',
            'Unit': 'Seconds'
        },
        {
            'AlarmName': 'RollbackRequired',
            'ComparisonOperator': 'GreaterThanThreshold',
            'EvaluationPeriods': 1,
            'MetricName': 'RollbackRequired',
            'Namespace': 'Restaurant/Deployment',
            'Period': 60,
            'Statistic': 'Sum',
            'Threshold': 0.0,
            'ActionsEnabled': True,
            'AlarmActions': [topic_arn],
            'AlarmDescription': 'Alert when rollback is triggered',
            'Unit': 'Count'
        }
    ]
    
    for alarm in alarms:
        cloudwatch.put_metric_alarm(**alarm)
        print(f"✅ Created alarm: {alarm['AlarmName']}")
    
    print(f"✅ SNS Topic created: {topic_arn}")
    print("✅ Monitoring setup complete!")

if __name__ == "__main__":
    setup_comprehensive_monitoring()
