Dependency Analyzer Lambda Function

# Lambda function: dependency-analyzer
import json
import boto3
from typing import Dict, List

def lambda_handler(event, context):
    services = event.get('services', [])
    strategy = event.get('strategy', 'parallel_optimized')
    
    # Analyze dependencies
    dependency_graph = build_dependency_graph(services)
    
    # Create optimized deployment plan
    deployment_plan = optimize_deployment_plan(dependency_graph, strategy)
    
    # Calculate estimated times
    time_estimation = calculate_deployment_times(deployment_plan)
    
    return {
        'dependency_graph': dependency_graph,
        'deployment_plan': deployment_plan,
        'time_estimation': time_estimation,
        'optimization_applied': True
    }

def build_dependency_graph(services: List[str]) -> Dict:
    """Build dependency graph for services"""
    dependencies = {
        'auth-service': [],
        'notification-service': [],
        'order-service': ['auth-service'],
        'payment-service': ['auth-service']
    }
    
    return {
        'services': services,
        'dependencies': {svc: dependencies.get(svc, []) for svc in services},
        'deployment_groups': create_deployment_groups(services, dependencies)
    }

def create_deployment_groups(services: List[str], dependencies: Dict) -> List[List[str]]:
    """Create groups of services that can be deployed in parallel"""
    groups = []
    remaining_services = services.copy()
    
    while remaining_services:
        # Find services with no unmet dependencies
        current_group = []
        for service in remaining_services.copy():
            service_deps = dependencies.get(service, [])
            if all(dep not in remaining_services for dep in service_deps):
                current_group.append(service)
                remaining_services.remove(service)
        
        if current_group:
            groups.append(current_group)
        else:
            # Circular dependency or error
            break
    
    return groups

def calculate_deployment_times(deployment_plan: Dict) -> Dict:
    """Calculate estimated deployment times"""
    service_times = {
        'auth-service': 90,      # seconds
        'notification-service': 60,
        'order-service': 120,
        'payment-service': 105
    }
    
    sequential_time = sum(service_times.values())
    
    # Calculate parallel time based on groups
    parallel_time = 0
    for group in deployment_plan['deployment_groups']:
        group_max_time = max(service_times.get(svc, 60) for svc in group)
        parallel_time += group_max_time
    
    return {
        'sequential': sequential_time,
        'parallel_optimized': parallel_time,
        'time_saved': sequential_time - parallel_time,
        'improvement_percentage': ((sequential_time - parallel_time) / sequential_time) * 100
    }

Scalability Tester Implementation

File: tests/scalability_tester.py

#!/usr/bin/env python3
import time
import boto3
import concurrent.futures
import json
from datetime import datetime

class ScalabilityTester:
    def __init__(self):
        self.stepfunctions = boto3.client('stepfunctions')
        self.state_machine_arn = "arn:aws:states:us-east-1:ACCOUNT:stateMachine:restaurant-deployment-orchestrator"
    
    def compare_strategies(self):
        """Compare deployment strategies performance"""
        strategies = ['sequential', 'parallel_optimized']
        results = {}
        
        for strategy in strategies:
            print(f"\n🧪 Testing {strategy} strategy...")
            
            start_time = time.time()
            execution_arn = self.start_deployment(strategy)
            self.wait_for_completion(execution_arn)
            end_time = time.time()
            
            duration = end_time - start_time
            results[strategy] = {
                'duration': duration,
                'execution_arn': execution_arn
            }
            
            print(f"✅ {strategy}: {duration:.1f} seconds")
        
        self.print_comparison(results)
        return results
    
    def concurrent_test(self, concurrent=3, duration=10):
        """Test concurrent deployments"""
        print(f"\n🔄 Testing {concurrent} concurrent deployments...")
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=concurrent) as executor:
            futures = []
            
            for i in range(concurrent):
                future = executor.submit(self.run_deployment_test, f"concurrent-{i}")
                futures.append(future)
            
            # Wait for all deployments
            results = []
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                results.append(result)
        
        self.analyze_concurrent_results(results)
        return results
    
    def start_deployment(self, strategy):
        """Start a deployment execution"""
        input_data = {
            'deployment_id': f'test-{strategy}-{int(time.time())}',
            'services': ['auth-service', 'notification-service', 'order-service', 'payment-service'],
            'strategy': strategy,
            'environment': 'test'
        }
        
        response = self.stepfunctions.start_execution(
            stateMachineArn=self.state_machine_arn,
            input=json.dumps(input_data)
        )
        
        return response['executionArn']
    
    def wait_for_completion(self, execution_arn):
        """Wait for execution to complete"""
        while True:
            response = self.stepfunctions.describe_execution(
                executionArn=execution_arn
            )
            
            status = response['status']
            if status in ['SUCCEEDED', 'FAILED', 'TIMED_OUT', 'ABORTED']:
                break
            
            time.sleep(5)
        
        return status
    
    def print_comparison(self, results):
        """Print strategy comparison"""
        print("\n📊 PERFORMANCE COMPARISON")
        print("=" * 50)
        
        sequential_time = results.get('sequential', {}).get('duration', 0)
        parallel_time = results.get('parallel_optimized', {}).get('duration', 0)
        
        if sequential_time and parallel_time:
            improvement = ((sequential_time - parallel_time) / sequential_time) * 100
            time_saved = sequential_time - parallel_time
            
            print(f"Sequential Strategy:     {sequential_time:.1f}s")
            print(f"Parallel Optimized:      {parallel_time:.1f}s")
            print(f"Time Saved:             {time_saved:.1f}s")
            print(f"Performance Improvement: {improvement:.1f}%")
        
        print("=" * 50)

if __name__ == "__main__":
    import sys
    
    tester = ScalabilityTester()
    
    if len(sys.argv) > 1:
        command = sys.argv[1]
        
        if command == "compare-strategies":
            tester.compare_strategies()
        elif command == "concurrent-test":
            concurrent = int(sys.argv[2]) if len(sys.argv) > 2 else 3
            tester.concurrent_test(concurrent=concurrent)
        else:
            print("Unknown command. Use: compare-strategies or concurrent-test")
    else:
        print("Usage: python scalability_tester.py [compare-strategies|concurrent-test]")
