10. Scalability Testing

Scalability Testing

Trong phần này, chúng ta sẽ thực hiện comprehensive scalability testing để đánh giá performance của hệ thống dưới các load conditions khác nhau.

Mục tiêu phần này:

  • Thiết lập load testing framework
  • Thực hiện concurrency testing
  • Analyze performance bottlenecks
  • Optimize system để handle production load

10.1 Load Testing Framework

Concurrent Execution Testing

Test Script Setup

// aws/testing/scalability-test.js
const AWS = require('aws-sdk');

const stepfunctions = new AWS.StepFunctions({
    region: 'us-east-1'
});

class ScalabilityTester {
    constructor(stateMachineArn) {
        this.stateMachineArn = stateMachineArn;
        this.results = [];
    }
    
    async runConcurrencyTest(concurrentExecutions = 10) {
        console.log(`🚀 Starting concurrency test with ${concurrentExecutions} parallel executions`);
        
        const startTime = Date.now();
        const promises = [];
        
        // Launch concurrent executions
        for (let i = 0; i < concurrentExecutions; i++) {
            const executionName = `concurrency-test-${i}-${Date.now()}`;
            const input = {
                testId: i,
                testType: 'concurrency',
                timestamp: new Date().toISOString()
            };
            
            promises.push(this.startExecution(executionName, input));
        }
        
        // Wait for all executions to complete
        const results = await Promise.allSettled(promises);
        const endTime = Date.now();
        
        // Analyze results
        const successful = results.filter(r => r.status === 'fulfilled').length;
        const failed = results.filter(r => r.status === 'rejected').length;
        
        const testResult = {
            testType: 'concurrency',
            concurrentExecutions,
            successful,
            failed,
            totalTime: endTime - startTime,
            averageTime: (endTime - startTime) / concurrentExecutions,
            successRate: (successful / concurrentExecutions) * 100
        };
        
        this.results.push(testResult);
        console.log('✅ Concurrency test completed:', testResult);
        
        return testResult;
    }
    
    async runLoadTest(totalExecutions = 50, batchSize = 10) {
        console.log(`🔥 Starting load test: ${totalExecutions} executions in batches of ${batchSize}`);
        
        const startTime = Date.now();
        const batches = Math.ceil(totalExecutions / batchSize);
        let totalSuccessful = 0;
        
        for (let batch = 0; batch < batches; batch++) {
            const batchStart = Date.now();
            const currentBatchSize = Math.min(batchSize, totalExecutions - (batch * batchSize));
            
            console.log(`📦 Processing batch ${batch + 1}/${batches} (${currentBatchSize} executions)`);
            
            const batchPromises = [];
            for (let i = 0; i < currentBatchSize; i++) {
                const executionName = `load-test-batch${batch}-${i}-${Date.now()}`;
                const input = {
                    testId: `${batch}-${i}`,
                    testType: 'load',
                    batchNumber: batch
                };
                
                batchPromises.push(this.startExecution(executionName, input));
            }
            
            const batchResults = await Promise.allSettled(batchPromises);
            const batchSuccessful = batchResults.filter(r => r.status === 'fulfilled').length;
            totalSuccessful += batchSuccessful;
            
            console.log(`✅ Batch ${batch + 1} completed: ${batchSuccessful}/${currentBatchSize} successful`);
            
            // Delay between batches
            if (batch < batches - 1) {
                await new Promise(resolve => setTimeout(resolve, 2000));
            }
        }
        
        const endTime = Date.now();
        const testResult = {
            testType: 'load',
            totalExecutions,
            totalSuccessful,
            totalTime: endTime - startTime,
            averageTime: (endTime - startTime) / totalExecutions,
            successRate: (totalSuccessful / totalExecutions) * 100,
            throughput: totalExecutions / ((endTime - startTime) / 1000)
        };
        
        this.results.push(testResult);
        console.log('✅ Load test completed:', testResult);
        
        return testResult;
    }
    
    async startExecution(name, input) {
        try {
            const params = {
                stateMachineArn: this.stateMachineArn,
                name,
                input: JSON.stringify(input)
            };
            
            const result = await stepfunctions.startExecution(params).promise();
            
            // Wait for completion (simplified polling)
            await this.waitForCompletion(result.executionArn);
            
            return result;
        } catch (error) {
            console.error(`❌ Execution failed: ${name}`, error);
            throw error;
        }
    }
    
    async waitForCompletion(executionArn, maxWaitTime = 60000) {
        const startTime = Date.now();
        
        while (Date.now() - startTime < maxWaitTime) {
            try {
                const result = await stepfunctions.describeExecution({
                    executionArn
                }).promise();
                
                if (result.status === 'SUCCEEDED' || result.status === 'FAILED') {
                    return result;
                }
                
                await new Promise(resolve => setTimeout(resolve, 1000));
            } catch (error) {
                console.error('Error checking execution status:', error);
                break;
            }
        }
        
        throw new Error('Execution timeout');
    }
    
    generateReport() {
        console.log('\n📊 SCALABILITY TEST REPORT');
        console.log('='.repeat(50));
        
        for (const result of this.results) {
            console.log(`\n🔍 ${result.testType.toUpperCase()} TEST:`);
            console.log(`   Success Rate: ${result.successRate.toFixed(2)}%`);
            console.log(`   Total Time: ${result.totalTime}ms`);
            console.log(`   Average Time: ${result.averageTime.toFixed(2)}ms`);
            
            if (result.throughput) {
                console.log(`   Throughput: ${result.throughput.toFixed(2)} executions/second`);
            }
        }
        
        return this.results;
    }
}

// Export for use
module.exports = ScalabilityTester;

// Demo execution
async function runDemo() {
    const stateMachineArn = 'arn:aws:states:us-east-1:ACCOUNT:stateMachine:restaurant-deployment-orchestrator';
    const tester = new ScalabilityTester(stateMachineArn);
    
    try {
        // Run concurrency test
        await tester.runConcurrencyTest(5);
        
        // Run load test  
        await tester.runLoadTest(20, 5);
        
        // Generate report
        tester.generateReport();
        
    } catch (error) {
        console.error('❌ Testing failed:', error);
    }
}

10.2 Performance Analysis

Key Performance Indicators

Throughput Metrics

  • Executions per second: Number of workflows completed per second
  • Concurrent execution capacity: Maximum simultaneous executions
  • Batch processing efficiency: Optimal batch size for throughput

Latency Metrics

  • Average execution time: Mean time for workflow completion
  • P95/P99 response times: Latency percentiles
  • Cold start impacts: Lambda initialization overhead

Reliability Metrics

  • Success rate under load: Percentage of successful executions
  • Error rate patterns: Types và frequency of errors
  • Recovery time after failures: Time to return to normal operation

10.3 Stress Testing Scenarios

Scenario 1: Peak Load Testing

Objective: Test system behavior under maximum expected load

async function runPeakLoadTest() {
    const tester = new ScalabilityTester(stateMachineArn);
    
    // Simulate peak deployment period
    const peakConfig = {
        totalExecutions: 100,
        batchSize: 20,
        concurrentBatches: 3
    };
    
    console.log('🔥 Starting peak load test...');
    
    const results = await tester.runLoadTest(
        peakConfig.totalExecutions,
        peakConfig.batchSize
    );
    
    // Analyze results
    if (results.successRate < 95) {
        console.warn('⚠️ Peak load test failed: Success rate below 95%');
        return false;
    }
    
    if (results.averageTime > 10000) {
        console.warn('⚠️ Peak load test failed: Average time exceeds 10 seconds');
        return false;
    }
    
    console.log('✅ Peak load test passed');
    return true;
}

Scenario 2: Sustained Load Testing

Objective: Test system stability over extended periods

async function runSustainedLoadTest(durationMinutes = 30) {
    const tester = new ScalabilityTester(stateMachineArn);
    const startTime = Date.now();
    const endTime = startTime + (durationMinutes * 60 * 1000);
    
    console.log(`🕐 Starting sustained load test for ${durationMinutes} minutes...`);
    
    let totalExecutions = 0;
    let totalSuccessful = 0;
    
    while (Date.now() < endTime) {
        try {
            const result = await tester.runLoadTest(10, 5);
            totalExecutions += result.totalExecutions;
            totalSuccessful += result.totalSuccessful;
            
            // Brief pause between test cycles
            await new Promise(resolve => setTimeout(resolve, 5000));
            
        } catch (error) {
            console.error('❌ Sustained load test error:', error);
        }
    }
    
    const sustainedResults = {
        testType: 'sustained',
        durationMinutes,
        totalExecutions,
        totalSuccessful,
        successRate: (totalSuccessful / totalExecutions) * 100,
        executionsPerMinute: totalExecutions / durationMinutes
    };
    
    console.log('✅ Sustained load test completed:', sustainedResults);
    return sustainedResults;
}

Scenario 3: Failure Recovery Testing

Objective: Test system recovery after failures

async function runFailureRecoveryTest() {
    const tester = new ScalabilityTester(stateMachineArn);
    
    console.log('🧪 Starting failure recovery test...');
    
    // Phase 1: Normal operation
    console.log('📊 Phase 1: Baseline performance');
    const baselineResult = await tester.runLoadTest(20, 5);
    
    // Phase 2: Inject failures
    console.log('💥 Phase 2: Simulating failures');
    const failureInput = {
        simulateFailure: {
            service: 'menu',
            errorType: 'timeout',
            probability: 0.3
        }
    };
    
    const failureResult = await tester.runLoadTest(20, 5, failureInput);
    
    // Phase 3: Recovery
    console.log('🔄 Phase 3: Recovery testing');
    const recoveryResult = await tester.runLoadTest(20, 5);
    
    // Analyze recovery
    const recoveryAnalysis = {
        baselineSuccessRate: baselineResult.successRate,
        failureSuccessRate: failureResult.successRate,
        recoverySuccessRate: recoveryResult.successRate,
        recoveryTime: recoveryResult.averageTime - baselineResult.averageTime
    };
    
    console.log('📈 Recovery analysis:', recoveryAnalysis);
    
    return recoveryAnalysis;
}

10.4 Performance Benchmarking

Benchmark Thresholds

Production Readiness Criteria:

  • Success rate: ≥ 99%
  • Average execution time: ≤ 30 seconds
  • P95 execution time: ≤ 45 seconds
  • Concurrent executions: ≥ 50
  • Throughput: ≥ 2 executions/second

Automated Benchmark Testing

class PerformanceBenchmark {
    constructor(stateMachineArn) {
        this.stateMachineArn = stateMachineArn;
        this.thresholds = {
            successRate: 99,
            averageTime: 30000,
            p95Time: 45000,
            maxConcurrency: 50,
            minThroughput: 2
        };
    }
    
    async runBenchmark() {
        console.log('🎯 Starting performance benchmark...');
        
        const tester = new ScalabilityTester(this.stateMachineArn);
        const results = [];
        
        // Test 1: Concurrency
        console.log('📊 Testing concurrency...');
        const concurrencyResult = await tester.runConcurrencyTest(this.thresholds.maxConcurrency);
        results.push(this.evaluateResult('concurrency', concurrencyResult));
        
        // Test 2: Throughput
        console.log('📊 Testing throughput...');
        const throughputResult = await tester.runLoadTest(100, 10);
        results.push(this.evaluateResult('throughput', throughputResult));
        
        // Test 3: Latency
        console.log('📊 Testing latency...');
        const latencyResult = await tester.runLoadTest(50, 5);
        results.push(this.evaluateResult('latency', latencyResult));
        
        // Generate benchmark report
        return this.generateBenchmarkReport(results);
    }
    
    evaluateResult(testType, result) {
        const evaluation = {
            testType,
            result,
            passed: true,
            issues: []
        };
        
        switch (testType) {
            case 'concurrency':
                if (result.successRate < this.thresholds.successRate) {
                    evaluation.passed = false;
                    evaluation.issues.push(`Success rate ${result.successRate}% below threshold ${this.thresholds.successRate}%`);
                }
                break;
                
            case 'throughput':
                if (result.throughput < this.thresholds.minThroughput) {
                    evaluation.passed = false;
                    evaluation.issues.push(`Throughput ${result.throughput} below threshold ${this.thresholds.minThroughput}`);
                }
                break;
                
            case 'latency':
                if (result.averageTime > this.thresholds.averageTime) {
                    evaluation.passed = false;
                    evaluation.issues.push(`Average time ${result.averageTime}ms exceeds threshold ${this.thresholds.averageTime}ms`);
                }
                break;
        }
        
        return evaluation;
    }
    
    generateBenchmarkReport(results) {
        const report = {
            timestamp: new Date().toISOString(),
            overallPassed: results.every(r => r.passed),
            results,
            summary: {
                totalTests: results.length,
                passed: results.filter(r => r.passed).length,
                failed: results.filter(r => !r.passed).length
            }
        };
        
        console.log('\n📊 BENCHMARK REPORT');
        console.log('='.repeat(50));
        console.log(`Overall Status: ${report.overallPassed ? '✅ PASSED' : '❌ FAILED'}`);
        console.log(`Tests: ${report.summary.passed}/${report.summary.totalTests} passed`);
        
        if (!report.overallPassed) {
            console.log('\n❌ Issues found:');
            results.forEach(result => {
                if (!result.passed) {
                    console.log(`\n${result.testType.toUpperCase()}:`);
                    result.issues.forEach(issue => console.log(`  - ${issue}`));
                }
            });
        }
        
        return report;
    }
}
