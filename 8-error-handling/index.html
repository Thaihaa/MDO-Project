async function simulateErrorScenario(simulateFailure, serviceName) {
    const { errorType, errorCode } = simulateFailure;
    
    console.log(`üß™ Simulating ${errorType} error for ${serviceName}`);
    
    switch (errorType) {
        case 'timeout':
            // Simulate timeout
            await new Promise(resolve => setTimeout(resolve, 35000));
            break;
            
        case 'service_unavailable':
            throw new Error(`Service ${serviceName} is temporarily unavailable`);
            
        case 'permission_denied':
            const permError = new Error(`Permission denied for ${serviceName}`);
            permError.code = 'PERMISSION_DENIED';
            permError.retryable = false;
            throw permError;
            
        case 'resource_exhausted':
            const resourceError = new Error(`Resource limit exceeded for ${serviceName}`);
            resourceError.code = 'RESOURCE_EXHAUSTED';
            resourceError.retryable = true;
            throw resourceError;
            
        default:
            throw new Error(`Unknown error type: ${errorType}`);
    }
}

function isRetryableError(error) {
    const retryableCodes = [
        'ECONNRESET',
        'ENOTFOUND', 
        'TIMEOUT',
        'RESOURCE_EXHAUSTED',
        'THROTTLING'
    ];
    
    return retryableCodes.includes(error.code) || error.retryable === true;
}

async function createRollbackPlan(failedService, deployedServices) {
    console.log(`üìã Creating rollback plan for failure: ${failedService}`);
    
    // Service dependency mapping
    const dependencies = {
        'review': ['order', 'auth'],
        'order': ['menu', 'auth'], 
        'menu': ['auth'],
        'restaurant': ['auth'],
        'auth': []
    };
    
    // Determine which services need rollback
    const servicesToRollback = [];
    
    // If auth fails, rollback everything
    if (failedService === 'auth') {
        servicesToRollback.push(...deployedServices);
    } else {
        // Rollback failed service and its dependents
        servicesToRollback.push(failedService);
        
        // Find dependent services
        for (const [service, deps] of Object.entries(dependencies)) {
            if (deps.includes(failedService) && deployedServices.includes(service)) {
                servicesToRollback.push(service);
            }
        }
    }
    
    // Order by reverse dependency (most dependent first)
    const rollbackOrder = ['review', 'order', 'restaurant', 'menu', 'auth']
        .filter(service => servicesToRollback.includes(service));
    
    const rollbackPlan = {
        failedService,
        services: rollbackOrder,
        strategy: 'reverse-dependency',
        phases: rollbackOrder.map((service, index) => ({
            phase: index + 1,
            service,
            action: 'rollback',
            dependencies: dependencies[service]
        }))
    };
    
    console.log(`üìã Rollback plan:`, JSON.stringify(rollbackPlan, null, 2));
    return rollbackPlan;
}

async function executeRollbackPlan(rollbackPlan) {
    console.log(`‚öôÔ∏è Executing rollback plan for ${rollbackPlan.services.length} services`);
    
    const results = [];
    
    for (const phase of rollbackPlan.phases) {
        console.log(`üîÑ Phase ${phase.phase}: Rolling back ${phase.service}`);
        
        try {
            const rollbackResult = await rollbackService(phase.service);
            
            results.push({
                phase: phase.phase,
                service: phase.service,
                status: 'ROLLED_BACK',
                duration: rollbackResult.duration,
                details: rollbackResult.operations,
                timestamp: new Date().toISOString()
            });
            
            // Wait between phases for stability
            if (phase.phase < rollbackPlan.phases.length) {
                console.log(`‚è≥ Waiting 2 seconds before next phase...`);
                await new Promise(resolve => setTimeout(resolve, 2000));
            }
            
        } catch (error) {
            console.error(`‚ùå Failed to rollback ${phase.service}:`, error);
            
            results.push({
                phase: phase.phase,
                service: phase.service,
                status: 'ROLLBACK_FAILED',
                error: error.message,
                timestamp: new Date().toISOString()
            });
            
            // Continue with other services despite failure
        }
    }
    
    return results;
}

async function validateRollback(services) {

    const validationResults = [];

                status: 'UNHEALTHY',

    }

}

8.5 Advanced Error Handling Patterns

Circuit Breaker Pattern

class CircuitBreaker {
    constructor(threshold = 5, timeout = 60000) {
        this.threshold = threshold;
        this.timeout = timeout;
        this.failureCount = 0;
        this.lastFailureTime = null;
        this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    }
    
    async execute(operation) {
        if (this.state === 'OPEN') {
            if (Date.now() - this.lastFailureTime > this.timeout) {
                this.state = 'HALF_OPEN';
            } else {
                throw new Error('Circuit breaker is OPEN');
            }
        }
        
        try {
            const result = await operation();
            this.onSuccess();
            return result;
        } catch (error) {
            this.onFailure();
            throw error;
        }
    }
    
    onSuccess() {
        this.failureCount = 0;
        this.state = 'CLOSED';
    }
    
    onFailure() {
        this.failureCount++;
        this.lastFailureTime = Date.now();
        
        if (this.failureCount >= this.threshold) {
            this.state = 'OPEN';
        }
    }
}
