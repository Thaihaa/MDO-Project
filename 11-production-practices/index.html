Secrets Management

// Using AWS Secrets Manager
const AWS = require('aws-sdk');
const secretsManager = new AWS.SecretsManager();

async function getSecret(secretId) {
    try {
        const result = await secretsManager.getSecretValue({
            SecretId: secretId
        }).promise();
        
        return JSON.parse(result.SecretString);
    } catch (error) {
        console.error('Failed to retrieve secret:', error);
        throw error;
    }
}

// Usage trong Lambda function
export const handler = async (event) => {
    const secrets = await getSecret('restaurant-app-secrets');
    const apiKey = secrets.apiKey;
    
    // Use API key securely
    const result = await makeApiCall(apiKey);
    return result;
};

11.2 Cost Optimization

Cost Analysis

Step Functions Pricing

  • Standard Workflows: $0.025 per 1,000 state transitions
  • Express Workflows: $1.00 per 1M requests + duration

Example Cost Calculation

Monthly Deployments: 1,000
Average States per Workflow: 12
Total State Transitions: 1,000 × 12 = 12,000

Cost = (12,000 / 1,000) × $0.025 = $0.30/month

Lambda Pricing

  • Requests: $0.20 per 1M requests
  • Duration: $0.0000166667 per GB-second

Cost Optimization Strategies

  1. Right-size Lambda functions: Optimize memory allocation
  2. Minimize state transitions: Combine related operations
  3. Use Express Workflows: For high-volume, short-duration workflows
  4. Implement caching: Reduce redundant operations

11.3 Operational Procedures

Deployment Checklist

Pre-Deployment

  • IAM roles và permissions verified
  • Lambda functions tested independently
  • State machine definition validated
  • Monitoring alerts configured
  • Rollback procedures documented

During Deployment

  • Monitor execution in real-time
  • Watch CloudWatch metrics
  • Verify each service health check
  • Confirm parallel execution performance
  • Validate error handling paths

Post-Deployment

  • Run end-to-end system validation
  • Check all monitoring dashboards
  • Verify alerting functionality
  • Update documentation
  • Conduct post-deployment review

Incident Response Procedures

Severity Levels

  • P0 - Critical: Complete system failure, immediate response required
  • P1 - High: Major functionality broken, 1-hour response
  • P2 - Medium: Partial functionality affected, 4-hour response
  • P3 - Low: Minor issues, next business day response

Response Playbook

# P0 Incident Response
1. Immediate Assessment:
   aws stepfunctions list-executions --state-machine-arn STATE_MACHINE_ARN --status-filter FAILED

2. Quick Rollback (if needed):
   aws stepfunctions start-execution \
     --state-machine-arn ROLLBACK_STATE_MACHINE_ARN \
     --name emergency-rollback-$(date +%s)

3. Notify Stakeholders:
   # Send alert to on-call engineer
   # Update status page
   # Inform stakeholders

4. Investigation:
   # Check CloudWatch logs
   # Review error patterns
   # Identify root cause

5. Resolution:
   # Implement fix
   # Test thoroughly  
   # Gradual rollout

11.6 Disaster Recovery

Backup Strategy

// State machine backup
async function backupStateMachine(stateMachineArn) {
    const stepfunctions = new AWS.StepFunctions();
    
    try {
        const definition = await stepfunctions.describeStateMachine({
            stateMachineArn
        }).promise();
        
        const backup = {
            timestamp: new Date().toISOString(),
            stateMachineArn,
            definition: definition.definition,
            roleArn: definition.roleArn,
            tags: definition.tags
        };
        
        // Store backup in S3
        await storeBackup(backup);
        
        console.log('✅ State machine backed up successfully');
        return backup;
        
    } catch (error) {
        console.error('❌ Backup failed:', error);
        throw error;
    }
}

Recovery Procedures

#!/bin/bash
# Disaster recovery script

# 1. Assess damage

11.7 Performance Optimization

Production Tuning

Performance Best Practices:

  • Monitor cold start impacts
  • Optimize Lambda memory allocation
  • Implement connection pooling
  • Use provisioned concurrency for critical functions
  • Cache frequently accessed data
// Production-optimized Lambda function
export const handler = async (event, context) => {
    // Optimize context
    context.callbackWaitsForEmptyEventLoop = false;
    
    // Connection pooling
    if (!global.connectionPool) {
        global.connectionPool = {
            http: new AWS.HttpClient({
                maxSockets: 50,
                keepAlive: true
            }),
            cache: new Map()
        };
    }
    
    // Caching strategy
    const cacheKey = `health-check-${event.serviceName}`;
    const cachedResult = global.connectionPool.cache.get(cacheKey);
    
    if (cachedResult && Date.now() - cachedResult.timestamp < 30000) {
        return cachedResult.result;
    }
    
    // Perform health check
    const result = await performHealthCheck(event);
    
    // Cache result
    global.connectionPool.cache.set(cacheKey, {
        result,
        timestamp: Date.now()
    });
    
    return result;
};

Monitoring Production Performance

// Performance monitoring
const performanceMetrics = {
    coldStarts: 0,
    warmStarts: 0,
    totalExecutions: 0,
    averageResponseTime: 0,
    errorRate: 0
};

function trackPerformance(startTime, isWarmStart, hasError) {
    const duration = Date.now() - startTime;
    
    performanceMetrics.totalExecutions++;
    
    if (isWarmStart) {
        performanceMetrics.warmStarts++;
    } else {
        performanceMetrics.coldStarts++;
    }
    
    if (hasError) {
        performanceMetrics.errorRate = 
            (performanceMetrics.errorRate * (performanceMetrics.totalExecutions - 1) + 1) / 
            performanceMetrics.totalExecutions;
    }
    
    performanceMetrics.averageResponseTime = 
        (performanceMetrics.averageResponseTime * (performanceMetrics.totalExecutions - 1) + duration) / 
        performanceMetrics.totalExecutions;
}
