Lambda Function Details

1. Deployment Initializer

Purpose: Khởi tạo deployment process và validate dependencies

# Key functionality
def lambda_handler(event, context):
    deployment_id = generate_deployment_id()
    services = event['services']
    
    # Validate service configurations
    for service in services:
        validate_service_config(service)
    
    # Create deployment plan
    plan = create_deployment_plan(services)
    
    return {
        'deployment_id': deployment_id,
        'plan': plan,
        'status': 'INITIALIZED'
    }

2. Microservice Deployer

Purpose: Deploy individual ECS services với health monitoring

def lambda_handler(event, context):
    service_name = event['service_name']
    
    # Deploy ECS service
    deployment = deploy_ecs_service(service_name)
    
    # Wait for stable state
    wait_for_stable_deployment(deployment)
    
    # Perform health check
    health_status = check_service_health(service_name)
    
    return {
        'service': service_name,
        'deployment_arn': deployment['deploymentArn'],
        'status': 'DEPLOYED',
        'health': health_status
    }

3. Health Checker

Purpose: Comprehensive health validation cho deployed services

def lambda_handler(event, context):
    service_name = event['service_name']
    
    # Check ECS task health
    ecs_health = check_ecs_health(service_name)
    
    # Check application endpoint
    app_health = check_endpoint_health(service_name)
    
    # Check dependencies
    deps_health = check_dependencies(service_name)
    
    overall_status = aggregate_health_status([
        ecs_health, app_health, deps_health
    ])
    
    return {
        'service': service_name,
        'health_status': overall_status,
        'details': {
            'ecs': ecs_health,
            'application': app_health,
            'dependencies': deps_health
        }
    }

4. Deployment Notifier

Purpose: Send notifications về deployment status

def lambda_handler(event, context):
    deployment_id = event['deployment_id']
    status = event['status']
    
    # Prepare notification message
    message = prepare_notification(deployment_id, status)
    
    # Send to SNS topic
    send_notification(message)
    
    # Log to CloudWatch
    log_deployment_event(deployment_id, status)
    
    return {
        'notification_sent': True,
        'timestamp': datetime.utcnow().isoformat()
    }
