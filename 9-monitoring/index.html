9.4 Structured Logging Strategy

Enhanced Logging Function

// Utility function for structured logging
function logEvent(level, event, context, metadata = {}) {
    const logEntry = {
        timestamp: new Date().toISOString(),
        level: level.toUpperCase(),
        event,
        context,
        metadata,
        requestId: context.awsRequestId || 'unknown',
        functionName: context.functionName || 'unknown'
    };
    
    console.log(JSON.stringify(logEntry));
}

// Usage trong Lambda functions
export const handler = async (event, context) => {
    logEvent('info', 'function_started', context, { 
        inputEvent: event,
        serviceName: event.serviceName 
    });
    
    try {
        const result = await processHealthCheck(event);
        
        logEvent('info', 'health_check_completed', context, {
            serviceName: event.serviceName,
            status: result.body.status,
            responseTime: result.body.responseTime
        });
        
        return result;
        
    } catch (error) {
        logEvent('error', 'health_check_failed', context, {
            serviceName: event.serviceName,
            errorMessage: error.message,
            errorStack: error.stack
        });
        
        throw error;
    }
};

Log Analysis Queries

CloudWatch Insights Queries

Query 1: Deployment Success Rate

fields @timestamp, level, event, metadata.serviceName, metadata.status
| filter event = "health_check_completed"
| stats count() as total, 
        count(metadata.status = "HEALTHY") as healthy
        by bin(5m)
| eval success_rate = (healthy / total) * 100

Query 2: Error Pattern Analysis

fields @timestamp, level, event, metadata.serviceName, metadata.errorMessage
| filter level = "ERROR"
| stats count() by metadata.serviceName, metadata.errorMessage
| sort count desc

Query 3: Performance Trends

fields @timestamp, metadata.serviceName, metadata.responseTime
| filter event = "health_check_completed"
| stats avg(metadata.responseTime) as avg_response_time by metadata.serviceName, bin(1h)

9.6 Distributed Tracing

X-Ray Integration

const AWSXRay = require('aws-xray-sdk-core');
const AWS = AWSXRay.captureAWS(require('aws-sdk'));

export const handler = async (event, context) => {
    const segment = AWSXRay.getSegment();
    const subsegment = segment.addNewSubsegment('health-check-operation');
    
    try {
        subsegment.addAnnotation('serviceName', event.serviceName);
        subsegment.addAnnotation('operation', 'health-check');
        
        const result = await performHealthCheck(event);
        
        subsegment.addMetadata('result', {
            status: result.body.status,
            responseTime: result.body.responseTime
        });
        
        subsegment.close();
        return result;
        
    } catch (error) {
        subsegment.addError(error);
        subsegment.close();
        throw error;
    }
};

9.7 Performance Monitoring

Key Performance Indicators

Monitoring Best Practices:

  • Set up proactive alerting
  • Monitor both technical v√† business metrics
  • Use correlation analysis ƒë·ªÉ identify root causes
  • Implement automated remediation where possible

Technical KPIs

  • Availability: Uptime percentage
  • Latency: Response time percentiles
  • Throughput: Requests per second
  • Error Rate: Percentage of failed requests

Business KPIs

  • Deployment Success Rate: Percentage of successful deployments
  • Mean Time to Recovery: Average time to fix issues
  • Service Dependency Health: Health of critical dependencies
  • Rollback Frequency: Number of rollbacks per time period

9.8 Troubleshooting Guide

Common Issues v√† Solutions

Issue 1: High Latency

Symptoms: Long execution times, timeout errors Investigation: Check CloudWatch metrics, Lambda cold starts Solution: Optimize Lambda memory, implement connection pooling

Issue 2: Intermittent Failures

Symptoms: Random execution failures Investigation: Check error patterns, dependency health Solution: Implement circuit breakers, improve retry logic

Issue 3: Resource Exhaustion

Symptoms: Throttling errors, capacity exceeded Investigation: Monitor concurrent executions, resource utilization Solution: Implement backpressure, scale resources

Monitoring Checklist

Production Readiness:

  • CloudWatch dashboards configured
  • Critical alerts set up
  • Log aggregation working
  • Custom metrics implemented
  • Runbooks documented
  • On-call procedures defined
